\subsubsection*{Taylor Polynomial}
$$P_n(x) = \sum_{k = 0}^n \frac{f^{(k)}(c)}{k!} (x - c)^k$$

\subsubsection*{Taylor's Remainder Formula}
$$f(x) - P_n(x) = \frac{f^{(n+1)}(z)}{(n+1)!} (x-c)^{n+1} \text{ for some } z \text{ between } x ~\&~ c$$

\subsubsection*{Newton's Method}
$$x_{n+1} = x_n - \dfrac{f(x_n)}{f'(x_n)}$$

\subsubsection*{Secant Method}
$$x_{n+1} = x_n - \dfrac{f(x_n) (x_n - x_{n-1})}{f(x_n) - f(x_{n-1})}$$


\subsubsection*{Newton's Method for Systems} ~
$$\mathbf{x}_{n+1} = \mathbf{x}_n - \mathbf{J}(\mathbf{x}_n)^{-1} \mathbf{F}(\mathbf{x}_n)$$

\subsubsection*{Condition Number} ~
$$\kappa(A) = \| A \| \, \|A^{-1} \|$$
