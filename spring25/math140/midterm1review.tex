\question Simplify by factoring $\ds \frac{x^2-5x+6}{4x-8}$.
\begin{solution}
$$\frac{x^2-5x+6}{4x^2-8} = \frac{(x-2)(x-3)}{4(x-2)} = \frac{x-3}{4}$$
\end{solution}
\vfill
\vfill
\vfill

\question Simplify $(\sqrt{2} + \sqrt{50})^2$. 
\begin{solution}
$$(\sqrt{2}+\sqrt{50})^2 = 2 + 2 \sqrt{2}\sqrt{50} + 50 = 52 + 2 \sqrt{100} = 72$$
\end{solution}
\vfill
